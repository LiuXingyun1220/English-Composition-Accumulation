# 5. 爪型行列式 -> 三角型行列式
$$
\begin{aligned}
D_{n} &= \begin{vmatrix}
\lambda_{0} & a_{1} & a_{2} & \dots & a_{n} \\
b_{1} & \lambda_{1} & 0 & \dots & 0 \\
b_{2} & 0 & \lambda_{2} & \dots & 0 \\
\vdots & \vdots & \vdots & \vdots & \vdots \\
b_{n} & 0 & 0 & \dots & \lambda_{n}
\end{vmatrix} (\lambda_{1} \lambda_{2}\dots \lambda_{n} \neq 0)\\
      &= \left( \lambda_{0}-\sum_{i=1}^{n} \frac{a_{i} b_{i}}{\lambda_{i}} \right) \lambda_{1} \lambda_{2}\dots \lambda_{n}
\end{aligned}
$$

# 8. 矩阵分块行列式
$$
\begin{vmatrix}
A & O \\
O & B
\end{vmatrix}=
\begin{vmatrix}
A & C \\
O & B
\end{vmatrix}=
\begin{vmatrix}
A & O \\
D & B
\end{vmatrix}=|A||B|
$$
$$
\begin{vmatrix}
O & A_{n \times n} \\
B_{m\times m} & O
\end{vmatrix}=
\begin{vmatrix}
O & A \\
B & C
\end{vmatrix}=
\begin{vmatrix}
D & A \\
B & O
\end{vmatrix}=(-1)^{mn}|A||B|
$$
# 9. 范德蒙行列式
$$
V_{n}=\begin{vmatrix}
1 & 1 & \dots & 1 \\
x_{1} & x_{2} & \dots & x_{n} \\
x_{1}^{2} & x_{2}^{2} & \dots & x_{n}^{2} \\
\vdots & \vdots & \ddots & \vdots \\
x_{1}^{n} & x_{2}^{n} & \dots & x_{n}^{n} 
\end{vmatrix}=\prod_{n\geq i> j\geq1}(x_{i}-x_{j})
$$
